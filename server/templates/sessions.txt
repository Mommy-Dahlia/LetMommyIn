    <script>
      let planObj = { plan: [] };

      function readPlanFromTextarea() {
        const txt = document.getElementById("plan_json").value.trim();
        if (!txt) return { plan: [] };
        return JSON.parse(txt);
      }

      function writePlanToTextarea() {
        document.getElementById("plan_json").value = JSON.stringify(planObj, null, 2);
      }

      function renderPlanList() {
        const ol = document.getElementById("planList");
        ol.innerHTML = "";
        (planObj.plan || []).forEach((item, idx) => {
          const li = document.createElement("li");
          li.style.marginBottom = "6px";

          const row = document.createElement("div");
          row.className = "row";

          const up = document.createElement("button");
          up.type = "button";
          up.textContent = "↑";
          up.onclick = () => moveItem(idx, -1);

          const down = document.createElement("button");
          down.type = "button";
          down.textContent = "↓";
          down.onclick = () => moveItem(idx, +1);

          const del = document.createElement("button");
          del.type = "button";
          del.textContent = "Delete";
          del.onclick = () => { planObj.plan.splice(idx, 1); sync(); };

          row.appendChild(up); row.appendChild(down); row.appendChild(del);

          const pre = document.createElement("pre");
          pre.textContent = JSON.stringify(item, null, 2);
          pre.style.margin = "0";

          li.appendChild(row);
          li.appendChild(pre);
          ol.appendChild(li);
        });
      }

      function moveItem(idx, delta) {
        const j = idx + delta;
        if (j < 0 || j >= planObj.plan.length) return;
        const tmp = planObj.plan[idx];
        planObj.plan[idx] = planObj.plan[j];
        planObj.plan[j] = tmp;
        sync();
      }

      function sync() {
        writePlanToTextarea();
        renderPlanList();
      }

      function addLinesPrompt() {
        const raw = prompt("Paste lines (each newline becomes one line).");
        if (!raw) return;
        const lines = raw.replace(/\r\n/g, "\n").replace(/\r/g, "\n").split("\n");
        planObj.plan.push({ lines: lines });
        sync();
      }

      function togglePick(btn) {
        btn.classList.toggle("selected");
        btn.textContent = btn.classList.contains("selected") ? "Selected" : "Select";
      }

      function clearSelected() {
        document.querySelectorAll(".block-pick.selected").forEach(btn => {
          btn.classList.remove("selected");
          btn.textContent = "Select";
        });
      }

      function getSelectedTitles() {
        return Array.from(document.querySelectorAll(".block-pick.selected"))
          .map(btn => btn.dataset.title)
          .filter(Boolean);
      }

      function addSelectedBlocks() {
        const titles = getSelectedTitles();
        if (titles.length === 0) {
          alert("No blocks selected.");
          return;
        }

        if (titles.length === 1) {
          planObj.plan.push({ include: titles[0] });
        } else {
          // default choose policy: pick 1 of N (you can tweak later)
          planObj.plan.push({ choose: { from: titles, min: 1, max: 1 } });
        }

        clearSelected();
        sync();
      }

      function clearPlan() {
        planObj = { plan: [] };
        sync();
      }

      (function init() {
        const txt = document.getElementById("plan_json").value.trim();
        if (txt) {
          try { planObj = JSON.parse(txt); }
          catch { planObj = { plan: [] }; writePlanToTextarea(); }
        } else {
          writePlanToTextarea();
        }
        renderPlanList();

        document.getElementById("plan_json").addEventListener("blur", () => {
          try {
            planObj = readPlanFromTextarea();
            if (!planObj.plan) planObj.plan = [];
            renderPlanList();
          } catch (e) {}
        });
      })();
    </script>
