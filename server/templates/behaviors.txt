    <script>
      function loadEntry(type, name) {
        fetch(`/admin/behaviors/load?name=${encodeURIComponent(name)}&behavior_type=${encodeURIComponent(type)}`)
          .then(r => r.json())
          .then(data => {
            if (type === 'toys_and_teases') {
              document.getElementById('toys-name').value = data.name;
              // reconstruct script from messages array
              const lines = (data.entry.messages || [])
                .map(m => `${m.text} #${m.delay_seconds}`)
                .join('\n');
              document.getElementById('toys-script').value = lines;
            } else if (type === 'rules_and_tasks') {
              document.getElementById('rules-name').value = data.name;
              document.getElementById('rules-task').value = data.entry.task || '';
              document.getElementById('rules-check').value = data.entry.check_text || '';
              document.getElementById('rules-timer').value = data.entry.timer_minutes || 5;
              document.getElementById('rules-reward').value = data.entry.reward || '';
              document.getElementById('rules-punishment').value = data.entry.punishment || '';
            } else if (type === 'web_aided_tasks') {
              document.getElementById('web-name').value = data.name;
              document.getElementById('web-url').value = data.entry.url || '';
              document.getElementById('web-message').value = data.entry.message || '';
            }
          });
      }
    </script>
